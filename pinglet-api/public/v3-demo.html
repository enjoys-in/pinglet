API Reference

POST /api/v1/notifications/send — Type 0 (Glassmorphism)

{
  "projectId": "YOUR_24_CHAR_PROJECT_ID",
  "type": "0",
  "body": {
    "title": "Flash Sale 🔥",
    "description": "50% off all items.",
    "icon": "https://cdn.example.com/icon.png",
    "media": { "type": "image", "src": "https://..." },
    "buttons": [
      { "text": "Shop", "action": "redirect", "src": "https://..." },
      { "text": "Track", "action": "event", "event": "sale:click" },
      { "text": "Close", "action": "close" }
    ]
  }
}

Premium Overrides (any type)

{
  "overrides": {
    "position": "top-right",
    "transition": "slide",
    "duration": 8000,
    "maxVisible": 5,

    "theme": { "mode": "dark" },
    "branding": { "show": true, "html": "Powered by Acme" },
    "sound": { "play": true, "src": "https://...", "volume": 0.7 }
  }
}
